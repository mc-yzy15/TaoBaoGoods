# 淘宝自动购买器配置文件
# 说明：请根据实际情况修改以下配置

# 浏览器窗口大小
window_size=1280,720

# 代理服务器（可选，留空表示不使用代理）
proxy=

# ChromeDriver路径
# 注意：请确保ChromeDriver版本与您的Chrome浏览器版本匹配
# 下载地址：https://chromedriver.chromium.org/downloads
driver_path=chromedriver.exe

# 淘宝账号信息
username=your_username
password=your_password

# 商品列表
# 格式：item=商品URL:购买数量
item=https://example.com/item1:1
item=https://example.com/item2:2

# 使用说明：
# 1. 运行build.bat编译程序
# 2. 确保chromedriver.exe在同一目录下
# 3. 运行main.exe开始自动购买流程
# 4. 程序会自动打开浏览器并引导您完成操作
